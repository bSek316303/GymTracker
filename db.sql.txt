-- 1. Tabela Użytkowników (Podstawa, do której przypisujemy resztę danych)
CREATE TABLE users (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(32) NOT NULL UNIQUE,
    email VARCHAR(64) NOT NULL UNIQUE,
    password VARCHAR(80) NOT NULL,
);

-- 2. Tabela Historii Wag (Relacja Wiele-do-Jednego z users)
CREATE TABLE weights (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    weight_date DATE NOT NULL,
    weight FLOAT NOT NULL,
    user_id BIGINT NOT NULL,

    CONSTRAINT fk_weights_user FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

-- 3. Tabela Wymiarów Ciała (Relacja Wiele-do-Jednego z users)
CREATE TABLE body_measurements (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    measurement_date DATE NOT NULL,
    chest FLOAT NULL,
    shoulders FLOAT NULL,
    biceps FLOAT NULL,
    thigh FLOAT NULL,
    waist FLOAT NULL,
    belly FLOAT NULL,
    hips FLOAT NULL,
    user_id BIGINT NOT NULL,

    CONSTRAINT fk_measurements_user FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE workout_plans (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    plan_name VARCHAR(32) NOT NULL,
    days_per_week INT NOT NULL,
    user_id BIGINT NOT NULL,
    CONSTRAINT fk_plans_user FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE plan_exercises (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    exercise_name VARCHAR(64) NOT NULL,
    target_sets INT NOT NULL,
    notes VARCHAR(255) NULL,
    day_number INT NOT NULL,
    exercise_number INT NOT NULL,
    plan_id BIGINT NOT NULL,
    CONSTRAINT fk_exercises_plan FOREIGN KEY (plan_id) REFERENCES workout_plans(id) ON DELETE CASCADE
);

CREATE TABLE trainings (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    plan_id BIGINT,
    day_number INT NOT NULL,
    training_date DATE NOT NULL,
    training_week INT NOT NULL,

    CONSTRAINT fk_trainings_workout FOREIGN KEY (plan_id) REFERENCES workout_plans(id) ON DELETE CASCADE
);

create table exercise_results (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    training_id BIGINT,
    exercise_id BIGINT,

    CONSTRAINT fk_exercise_results_trainings FOREIGN KEY (training_id) REFERENCES trainings(id) ON DELETE CASCADE
);

CREATE TABLE set_logs (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    exercise_result_id BIGINT NOT NULL,
    set_number INT NOT NULL,

    weight DECIMAL(5,2) NULL,
    reps INT NULL,
    duration_seconds INT NULL,
    distance_meters DECIMAL(7,2) NULL,
    calories INT NULL,

    CONSTRAINT fk_set_logs_exercise_results FOREIGN KEY (exercise_result_id) REFERENCES exercise_results(id) ON DELETE CASCADE
);
